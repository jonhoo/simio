#
# Clock IO automaton
#

Name:
	Clock

Signature-Input:
	request

Signature-Internal:
	tick

Signature-Output:
	clock(t)

# XXX -- its lame to make the user prepend "self."s everywhere...
State:
	self.counter = 0
	self.flag = False

Transition-Name:
	tick
Transition-Precondition:
	return True
Transition-Effect:
	self.counter += 1

Transition-Name:
	request
Transition-Effect:
	self.flag = True

Transition-Name:
	clock(t)
Transition-Precondition:
	return self.flag == True
Transition-Output:
	return self.counter
Transition-Effect:
	self.flag = False

Tasks:
	tick
	clock(t)
